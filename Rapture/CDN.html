def application_asset_url(id, asset_id, ext: "png", size: nil)
  check_extension!(%w[png jpeg webp], ext)
  check_size!(size) if size

  "#{BASE}/app-assets/#{id}/#{asset_id}.#{ext}" + get_query(size)
end

end
